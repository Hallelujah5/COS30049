CREATE TABLE User (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(255) NOT NULL,
    user_wallet VARCHAR(255) NOT NULL,
    date_join DATE NOT NULL
);

CREATE TABLE NFT (
    nft_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    own_by INT,
    current_price DECIMAL(10, 2) NOT NULL,
    description TEXT,
    details TEXT,
    FOREIGN KEY (own_by) REFERENCES User(user_id)
);

CREATE TABLE Transactions (
    transac_id INT AUTO_INCREMENT PRIMARY KEY,
    nft_id INT,
    from_address INT,
    to_address INT,
    time DATETIME NOT NULL,
    price DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (nft_id) REFERENCES NFT(nft_id),
    FOREIGN KEY (from_address) REFERENCES User(user_id),
    FOREIGN KEY (to_address) REFERENCES User(user_id)
);
